235  void handleEvent(sf::RenderWindow& Window, sf::Event& event) override
236  {
237  sf::Vector2i pos = sf::Mouse::getPosition(Window);
238  if (event.type == sf::Event::Closed)
239  Window.close();
240  else if (event.type == sf::Event::MouseButtonPressed)
241  {
242  if (event.mouseButton.button == sf::Mouse::Left)
243  {
244  if (buttons[0]->mousePress(pos)) {
245  }
246  else if (buttons[1]->mousePress(pos)) {
247  }
248  else if (sf::Mouse::getPosition().y - Window.getPosition().y < 35) {
249  m_windowPossChange = Window.getPosition() - sf::Mouse::getPosition();
250  m_grabbedWindow = true;
251  }
252  }
253  }
254  else if (event.type == sf::Event::MouseButtonReleased)
255  {
256  if (event.mouseButton.button == sf::Mouse::Left) {
257  if (buttons[0]->mouseRelease(pos)) {
258  Window.close();
259  }
260  else if (buttons[1]->mouseRelease(pos)) {
261  PLATFORM::minimalize(Window.getSystemHandle());
262  buttons[1]->forceRelease();
263  }
264  m_grabbedWindow = false;
265  }
266  }
267  else if (event.type == sf::Event::MouseMoved)
268  {
269  if (m_grabbedWindow)
270  Window.setPosition(sf::Mouse::getPosition() + m_windowPossChange);
271  else {
272  buttons[0]->mouseActivity(pos);
273  buttons[1]->mouseActivity(pos);
274  }
275 
276  }
277  else if (event.type == sf::Event::GainedFocus) {
278  PLATFORM::maximalize(Window.getSystemHandle());
279  }
280  additionalEvents(Window, event);
281  }
282  };
