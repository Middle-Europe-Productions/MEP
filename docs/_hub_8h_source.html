236  void handleEvent(sf::RenderWindow& Window, sf::Event& event) override
237  {
238  sf::Vector2i pos = sf::Mouse::getPosition(Window);
239  if (event.type == sf::Event::Closed)
240  Window.close();
241  else if (event.type == sf::Event::MouseButtonPressed)
242  {
243  if (event.mouseButton.button == sf::Mouse::Left)
244  {
245  if (buttons[0]->mousePress(pos)) {
246  }
247  else if (buttons[1]->mousePress(pos)) {
248  }
249  else if (sf::Mouse::getPosition().y - Window.getPosition().y < 35) {
250  m_windowPossChange = Window.getPosition() - sf::Mouse::getPosition();
251  m_grabbedWindow = true;
252  }
253  }
254  }
255  else if (event.type == sf::Event::MouseButtonReleased)
256  {
257  if (event.mouseButton.button == sf::Mouse::Left) {
258  if (buttons[0]->mouseRelease(pos)) {
259  Window.close();
260  }
261  else if (buttons[1]->mouseRelease(pos)) {
262  PLATFORM::minimalize(Window.getSystemHandle());
263  buttons[1]->forceRelease();
264  }
265  m_grabbedWindow = false;
266  }
267  }
268  else if (event.type == sf::Event::MouseMoved)
269  {
270  if (m_grabbedWindow)
271  Window.setPosition(sf::Mouse::getPosition() + m_windowPossChange);
272  else {
273  buttons[0]->mouseActivity(pos);
274  buttons[1]->mouseActivity(pos);
275  }
276 
277  }
278  else if (event.type == sf::Event::GainedFocus) {
279  PLATFORM::maximalize(Window.getSystemHandle());
280  }
281  additionalEvents(Window, event);
282  }
283  };
284  }
