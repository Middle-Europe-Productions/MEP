90  void MEP::AnimationPossition::init()
91  {
92  double jump = (m_end - m_begin) / nofFrames;
93  double min = sizeof(double), max = 0;
94  for (double x = m_begin; x < m_end; x += jump) {
95  frames.push_back(m_function(x));
96  if (frames.back() > max)
97  max = frames.back();
98  if (frames.back() < min)
99  min = frames.back();
100  }
101  if (min < 0)
102  for (auto& x : frames)
103  x += abs(min);
104  double multiplication = (m_exit - m_entry) / (max - min);
105  for (auto& x : frames) {
106  x = (m_entry + ((x)*multiplication));
107  }
108  for (auto& x : frames) {
109  std::cout << x << std::endl;

111  currentFrame = frames.begin();

122  void MEP::AnimationPossition::update(sf::Time& currentTime)
123  {
124  if (isRunning and (isInit == AnimationInit::PossitionAnimation)) {
125  if (currentTime - updateTime >= toWait) {
126  if (direction == Direction::Forward) {
127  if (currentFrame == --frames.end()) {
128  isRunning = false;
129  }
130  else {
131  currentFrame++;
132  current_frame++;
133  }
134  }
135  else if (direction == Direction::Backwards) {
136  if (currentFrame == frames.begin()) {
137  isRunning = false;
138  }
139  else {
140  currentFrame--;
141  current_frame--;
142  }
143  }
144  updateTime = currentTime;
145  }
146  }
147  else if (m_delay != sf::Time::Zero) {
148  if (!delayInit) {
149  updateTime = currentTime;
150  delayInit = true;
151  }
152  else if (currentTime - updateTime > m_delay) {
153  Run(direction);
154  m_delay = sf::Time::Zero;
155  delayInit = false;
156  }
157  }
