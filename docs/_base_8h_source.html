394  inline void BaseWindow::handleEvent(sf::RenderWindow& Window, sf::Event& event)
395  {
396  if (event.type == sf::Event::Closed)
397  Window.close();
398  else if (event.type == sf::Event::MouseButtonPressed)
399  {
400  if (event.mouseButton.button == sf::Mouse::Left)
401  {
402  m_windowPossChange = Window.getPosition() - sf::Mouse::getPosition();
403  m_grabbedWindow = true;
404  }
405  }
406  else if (event.type == sf::Event::MouseButtonReleased)
407  {
408  if (event.mouseButton.button == sf::Mouse::Left)
409  m_grabbedWindow = false;
410  }
411  else if (event.type == sf::Event::MouseMoved)
412  {
413  if (m_grabbedWindow)
414  Window.setPosition(sf::Mouse::getPosition() + m_windowPossChange);
415  }
416  }
