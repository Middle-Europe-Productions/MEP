349  inline void BaseWindow::update(sf::Time& currentTime)
350  {
351  try {
352  beforeUpdate(currentTime);
353  if (m_status == Status::InProgress or m_status == Status::Main) {
354  updateRunning(currentTime);
355  }
356  else if (m_status == Status::Entrance) {
357  updateEntrance(currentTime);
358  }
359  else if (m_status == Status::Exit) {
360  updateExit(currentTime);
361  }
362  afterUpdate(currentTime);
363  }
364  catch (const MEP::Window::WindowException& x) {
365  throw x;
366  }
367  catch (const MEP::ResourceException& x) {
368  throw x;
369  }
370  }
