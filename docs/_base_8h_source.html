331  inline void BaseWindow::handleEvent(sf::RenderWindow& Window, sf::Event& event)
332  {
333  if (event.type == sf::Event::Closed)
334  Window.close();
335  else if (event.type == sf::Event::MouseButtonPressed)
336  {
337  if (event.mouseButton.button == sf::Mouse::Left)
338  {
339  m_windowPossChange = Window.getPosition() - sf::Mouse::getPosition();
340  m_grabbedWindow = true;
341  }
342  }
343  else if (event.type == sf::Event::MouseButtonReleased)
344  {
345  if (event.mouseButton.button == sf::Mouse::Left)
346  m_grabbedWindow = false;
347  }
348  else if (event.type == sf::Event::MouseMoved)
349  {
350  if (m_grabbedWindow)
351  Window.setPosition(sf::Mouse::getPosition() + m_windowPossChange);
352  }
353  }
