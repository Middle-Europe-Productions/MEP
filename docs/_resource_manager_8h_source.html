332  inline void Resources::load(const MEP::Resource& data)
333  {
334  if (data.m_type == ResourceType::Multi) {
335  try {
336  //finds the target group
337  std::list<std::unique_ptr<Object>>& x = findOrAdd<MEP::Object, Group<MEP::Object>>(data.m_group, objects);
338  //check is the resource already exists in a group
339  check<Object>(x, data.m_ID);
340  x.push_back(std::make_unique<MEP::Object>(data.m_ID, m_path, data.m_name, data.m_nofFrames, data.m_transparency));
341  }
342  catch (const char* x) {
343  throw ResourceException(data.m_name, x, ResourceException::ExceptionType::CouldntLoad);
344  }
345  }
346  else if (data.m_type == ResourceType::Single) {
347  try {
348  //finds the target group
349  std::list<std::unique_ptr<Object>>& x = findOrAdd<MEP::Object, Group<MEP::Object>>(data.m_group, objects);
350  //check is the resource already exists in a group
351  check<Object>(x, data.m_ID);
352  x.push_back(std::make_unique<MEP::Object>(data.m_ID, m_path, data.m_name, data.m_transparency ));
353  }
354  catch (const char* x) {
355  throw ResourceException(data.m_name, x, ResourceException::ExceptionType::CouldntLoad);
356  }
357  }
358  else if (data.m_type == ResourceType::ImageArray) {
359  try {
360  //finds the target group
361  std::list<std::unique_ptr<Object>>& x = findOrAdd<MEP::Object, Group<MEP::Object>>(data.m_group, objects);
362  //check is the resource already exists in a group
363  check<Object>(x, data.m_ID);
364  x.push_back(std::make_unique<MEP::Object>(data.m_ID, *data.m_array, data.m_name, data.m_transparency ));
365  }
366  catch (const char* x) {
367  throw ResourceException(data.m_name, x, ResourceException::ExceptionType::CouldntLoad);
368  }
369  }
370  else if (data.m_type == ResourceType::Font) {
371  //finds the target group
372  std::list<std::unique_ptr<Font>>& x = findOrAdd<Font, Group<Font>>(data.m_group, fonts);
373  //check is the resource already exists in a group
374  check<Font>(x, data.m_ID);
375  //initialize and add the newly created object.
376  x.push_back(std::make_unique<Font>(data.m_ID, data.m_name, m_path));
377  }
378  }
