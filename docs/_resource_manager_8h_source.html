247  inline void Resources::load(const MEP::Resource& data)
248  {
249  if (data.m_type == ResourceType::Multi) {
250  try {
251  if (!MEPtools::GroupManager<MEP::Object>::_insert(data.m_ID,
252  data.m_group,
253  std::make_unique<MEP::Object>(data.m_ID, m_path, data.m_name, data.m_nofFrames, data.m_transparency))) {
254  throw ResourceException(data.m_name, "Resource already exists in that group!", ResourceException::ExceptionType::ResourceAlreadyExists);
255  }
256  }
257  catch (const char* x) {
258  throw ResourceException(data.m_name, x, ResourceException::ExceptionType::CouldntLoad);
259  }
260  }
261  else if (data.m_type == ResourceType::Single) {
262  try {
263  if (!MEPtools::GroupManager<MEP::Object>::_insert(data.m_ID,
264  data.m_group,
265  std::make_unique<MEP::Object>(data.m_ID, m_path, data.m_name, data.m_transparency))) {
266  throw ResourceException(data.m_name, "Resource already exists in that group!", ResourceException::ExceptionType::ResourceAlreadyExists);
267  }
268  }
269  catch (const char* x) {
270  throw ResourceException(data.m_name, x, ResourceException::ExceptionType::CouldntLoad);
271  }
272  }
273  else if (data.m_type == ResourceType::ImageArray) {
274  try {
275  if(!MEPtools::GroupManager<MEP::Object>::_insert(data.m_ID,
276  data.m_group,
277  std::make_unique<MEP::Object>(data.m_ID, *data.m_array, data.m_name, data.m_transparency))) {
278  throw ResourceException(data.m_name, "Resource already exists in that group!", ResourceException::ExceptionType::ResourceAlreadyExists);
279  }
280  }
281  catch (const char* x) {
282  throw ResourceException(data.m_name, x, ResourceException::ExceptionType::CouldntLoad);
283  }
284  }
285  else if (data.m_type == ResourceType::Font) {
286  if (!MEPtools::GroupManager<MEP::Font>::_insert(data.m_ID,
287  data.m_group,
288  std::make_unique<MEP::Font>(data.m_ID, data.m_name, m_path))) {
289  throw ResourceException(data.m_name, "Resource already exists in that group!", ResourceException::ExceptionType::ResourceAlreadyExists);
290  }
291  }
292  }
