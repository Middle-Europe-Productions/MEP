303  inline void Resources::load(const MEP::Resources::Resource& data)
304  {
305  std::cout << data.m_name << " " << data.m_nofFrames << " " << (int)data.m_type << std::endl;
306  //finds the target group
307  std::list<std::unique_ptr<Object>>& x = findOrAdd(data.m_group);
308  //check is the resource already exists in a group
309  check(x, data.m_ID);
310  if (data.m_type == ResourceType::Multi) {
311  try {
312  x.push_back(std::make_unique<MEP::Object>(data.m_ID, m_path, data.m_name, data.m_nofFrames, data.m_transparency));
313  }
314  catch (const char* x) {
315  throw ResourceException(data.m_name, x, ResourceException::ExceptionType::CouldntLoad);
316  }
317  }
318  else if (data.m_type == ResourceType::Single) {
319  try {
320  x.push_back(std::make_unique<MEP::Object>(data.m_ID, m_path, data.m_name, data.m_transparency ));
321  }
322  catch (const char* x) {
323  throw ResourceException(data.m_name, x, ResourceException::ExceptionType::CouldntLoad);
324  }
325  }
326  else if (data.m_type == ResourceType::ImageArray) {
327  try {
328  x.push_back(std::make_unique<MEP::Object>(data.m_ID, *data.m_array, data.m_name, data.m_transparency ));
329  }
330  catch (const char* x) {
331  throw ResourceException(data.m_name, x, ResourceException::ExceptionType::CouldntLoad);
332  }
333  }
