213  inline void MEP::Object::loadTransparancy(sf::Image& surface)
214  {
215  //allocate table
216  table = new bool* [surface.getSize().x];
217  for (int i = 0; i < surface.getSize().x; ++i)
218  table[i] = new bool[surface.getSize().y];
219  //fill table
220  if (table) {
221  for (int i = 0; i < surface.getSize().x; i++) {
222  for (int j = 0; j < surface.getSize().y; j++) {
223  table[i][j] = surface.getPixel(i, j).a < 100;
224  }
225  }
226  transparency = true;
227  }
228  }

230  inline void MEP::Object::load(const std::string& fulladdress, bool transparencyM, bool masterSize) {
231  texture.push_back(new sf::Texture());
232  if (!texture.back()->loadFromFile(fulladdress)) {
233  throw "Texture not loaded!";
234  }
235  texture.back()->setSmooth(true);
236  if (transparencyM) {
237  sf::Image x;
238  if (!x.loadFromFile(fulladdress)) {
239  throw "Transparency mask not loaded!";
240  }
241  loadTransparancy(x);
242  }
243  if (masterSize) {
244  m_size.x = texture.front()->getSize().x;
245  m_size.y = texture.front()->getSize().y;
246  }
247  }
