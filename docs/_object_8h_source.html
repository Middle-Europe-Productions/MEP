225  inline void Object::loadTransparancy(sf::Image& surface)
226  {
227  //allocate table
228  table = new bool* [surface.getSize().x];
229  for (int i = 0; i < surface.getSize().x; ++i)
230  table[i] = new bool[surface.getSize().y];
231  //fill table
232  if (table) {
233  for (int i = 0; i < surface.getSize().x; i++) {
234  for (int j = 0; j < surface.getSize().y; j++) {
235  table[i][j] = surface.getPixel(i, j).a < 100;
236  }
237  }
238  transparency = true;
239  }
240  }

242  inline void Object::load(const std::string& fulladdress, bool transparencyM, bool masterSize) {
243  texture.push_back(new sf::Texture());
244  if (!texture.back()->loadFromFile(fulladdress)) {
245  throw "Texture not loaded!";
246  }
247  texture.back()->setSmooth(true);
248  if (transparencyM) {
249  sf::Image x;
250  if (!x.loadFromFile(fulladdress)) {
251  throw "Transparency mask not loaded!";
252  }
253  loadTransparancy(x);
254  }
255  if (masterSize) {
256  m_size.x = texture.front()->getSize().x;
257  m_size.y = texture.front()->getSize().y;
258  }
259  }
