347  inline void BaseWindow::update(sf::Time& currentTime)
348  {
349  try {
350  beforeUpdate(currentTime);
351  if (m_status == Status::InProgress or m_status == Status::Main) {
352  updateRunning(currentTime);
353  }
354  else if (m_status == Status::Entrance or m_status == Status::LowEntrance) {
355  updateEntrance(currentTime);
356  }
357  else if (m_status == Status::Exit or m_status == Status::LowExit) {
358  updateExit(currentTime);
359  }
360  afterUpdate(currentTime);
361  }
362  catch (const MEP::WindowException& x) {
363  throw x;
364  }
365  catch (const MEP::ResourceException& x) {
366  throw x;
367  }
368  }
