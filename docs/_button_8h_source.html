183  if (m_status == ButtonStatus::Active and index_currentFrame != m_active) {
184  if (index_currentFrame > m_active) {
185  while (index_currentFrame != m_active) {

221  inline bool MEP::Button::handleEvent(sf::Event& event, sf::Vector2i& pos)
222  {
223  if (m_block)
224  return false;
225  bool isTran = isTansparent(pos.x, pos.y);
226  if (event.type == sf::Event::MouseMoved or !followingListv2._empty()) {
227  if (!isTran)
228  if (sf::Mouse::isButtonPressed(sf::Mouse::Left))
229  changeStatus(ButtonStatus::Pressed);
230  else
231  if (event.type == sf::Event::MouseButtonReleased) {
232  changeStatus(ButtonStatus::Active);
233  return true;
234  }
235  else
236  changeStatus(ButtonStatus::Active);
237  else
238  changeStatus(ButtonStatus::Base);
239  }
240  else if (event.type == sf::Event::MouseButtonPressed) {
241  if (event.mouseButton.button == sf::Mouse::Left) {
242  if (!isTran)
243  changeStatus(ButtonStatus::Pressed);
244  }
245  }
246  else if (event.type == sf::Event::MouseButtonReleased) {
247  if (event.mouseButton.button == sf::Mouse::Left) {
248  if (!isTran) {
249  changeStatus(ButtonStatus::Active);
250  return true;
251  }
252  }
253  }
254  return false;
255  }
