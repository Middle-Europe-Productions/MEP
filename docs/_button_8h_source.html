202  bool MEP::Button::handleEvent(sf::Event& event, sf::Vector2i& pos)
203  {
204  bool isTran = isTansparent(pos.x, pos.y);
205  if (!followingList.empty()) {
206  if (!isTran)
207  if (sf::Mouse::isButtonPressed(sf::Mouse::Left))
208  changeStatus(ButtonStatus::Pressed);
209  else
210  changeStatus(ButtonStatus::Active);
211  else
212  changeStatus(ButtonStatus::Base);
213  }
214  if (event.type == sf::Event::MouseMoved) {
215  if (!isTran)
216  if (sf::Mouse::isButtonPressed(sf::Mouse::Left))
217  changeStatus(ButtonStatus::Pressed);
218  else
219  changeStatus(ButtonStatus::Active);
220  else
221  changeStatus(ButtonStatus::Base);
222  }
223  else if (event.type == sf::Event::MouseButtonPressed) {
224  if (event.key.code == sf::Mouse::Left) {
225  if (!isTran)
226  changeStatus(ButtonStatus::Pressed);
227  }
228  }
229  else if (event.type == sf::Event::MouseButtonReleased) {
230  if (event.key.code == sf::Mouse::Left) {
231  if (!isTran) {
232  changeStatus(ButtonStatus::Active);
233  return true;
234  }
235  }
236  }
237  return false;
