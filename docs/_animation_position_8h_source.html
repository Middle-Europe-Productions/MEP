172  inline void MEP::AnimationPosition::init()
173  {
174  double jump = (m_end - m_begin) / nofFrames;
175  double min = sizeof(double), max = 0;
176  for (double x = m_begin; x < m_end; x += jump) {
177  frames.push_back(m_function(x));
178  if (frames.back() > max)
179  max = frames.back();
180  if (frames.back() < min)
181  min = frames.back();
182  }
183  if (min < 0)
184  for (auto& x : frames)
185  x += abs(min);
186  double multiplication = (m_exit - m_entry) / (max - min);
187  for (auto& x : frames) {
188  x = (m_entry + ((x)*multiplication));
189  }
190  currentFrame = frames.begin();
191  }

209  inline void MEP::AnimationPosition::updatePositionAnimation(sf::Time& currentTime) {
210  if (isRunning and (isInit == AnimationInit::PositionAnimation)) {
211  if (currentTime - updateTime >= toWait) {
212  if (direction == Direction::Forward) {
213  if (currentFrame == --frames.end()) {
214  isRunning = false;
215  }
216  else {
217  currentFrame++;
218  current_frame++;
219  }
220  }
221  else if (direction == Direction::Backwards) {
222  if (currentFrame == frames.begin()) {
223  isRunning = false;
224  }
225  else {
226  currentFrame--;
227  current_frame--;
228  }
229  }
230  updateTime = currentTime;
231  }
232  }
233  else if (m_delay != sf::Time::Zero) {
234  if (!delayInit) {
235  updateTime = currentTime;
236  delayInit = true;
237  }
238  else if (currentTime - updateTime > m_delay) {
239  run(direction);
240  m_delay = sf::Time::Zero;
241  delayInit = false;
242  }
243  }
244  }
