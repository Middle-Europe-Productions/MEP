197  void MEP::AnimationObject::update(sf::Time& currentTime)
198  {
199  if (isRunning and (isInit == AnimationInit::ObjectAnimation)) {
200  if (currentTime - updateTime >= toWait) {
201  if (direction == Direction::Forward) {
202  if (currentFrame == --texture.end()) {
203  isRunning = false;
204  }
205  else {
206  currentFrame++;
207  index_currentFrame++;
208  }
209  }
210  else if (direction == Direction::Backwards) {
211  currentFrame--;
212  index_currentFrame++;
213  if (currentFrame == texture.begin()) {
214  isRunning = false;
215  }
216  }
217  updateTime = currentTime;
218  updateSprite();
219  }
220  }
221  if (isInit == AnimationInit::Follow) {
222  if (toFollowOBJ->GetInit() != AnimationInit::Follow) {
223  while (toFollowOBJ->index_currentFrame != index_currentFrame) {
224  if (toFollowOBJ->index_currentFrame > index_currentFrame) {
225  if (currentFrame == --texture.end()) {
226  break;
227  }
228  index_currentFrame++;
229  currentFrame++;
230  }
231  else {
232  if (currentFrame == texture.begin()) {
233  break;
234  }
235  index_currentFrame--;
236  currentFrame--;
237  }
238  }
239  if (toFollowOBJ->GetPosition() != GetPosition()) {
240  m_posFixed = toFollowOBJ->GetPosition();
241  }
242  if (toFollowOBJ->GetScale() != GetScale()) {
243  m_scaleFixed = toFollowOBJ->GetScale();
244  }
245  updateSprite();
246  }
247  }
248  if (!followingList.empty()) {
249  updateSprite();
250  }
251  }
