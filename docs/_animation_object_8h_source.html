164  void MEP::AnimationObject::update(sf::Time& currentTime)
165  {
166  if (isRunning and (isInit == AnimationInit::ObjectAnimation)) {
167  if (currentTime - updateTime >= toWait) {
168  if (direction == Direction::Forward) {
169  if (currentFrame == --texture.end()) {
170  isRunning = false;
171  }
172  else {
173  currentFrame++;
174  index_currentFrame++;
175  }
176  }
177  else if (direction == Direction::Backwards) {
178  currentFrame--;
179  index_currentFrame++;
180  if (currentFrame == texture.begin()) {
181  isRunning = false;
182  }
183  }
184  updateTime = currentTime;
185  updateSprite(**currentFrame);
186  }
187  }
188  if (isInit == AnimationInit::Follow) {
189  if (toFollowOBJ->getInit() != AnimationInit::Follow) {
190  while (toFollowOBJ->index_currentFrame != index_currentFrame) {
191  if (toFollowOBJ->index_currentFrame > index_currentFrame) {
192  if (currentFrame == --texture.end()) {
193  break;
194  }
195  index_currentFrame++;
196  currentFrame++;
197  }
198  else {
199  if (currentFrame == texture.begin()) {
200  break;
201  }
202  index_currentFrame--;
203  currentFrame--;
204  }
205  }
206  if (toFollowOBJ->getPosition() != getPosition()) {
207  m_posFixed = toFollowOBJ->getPosition();
208  }
209  if (toFollowOBJ->getScale() != getScale()) {
210  m_scaleFixed = toFollowOBJ->getScale();
211  }
212  updateSprite(**currentFrame);
213  }
214  }
215  if (!followingList.empty()) {
216  updateSprite();
217  }
218  }
