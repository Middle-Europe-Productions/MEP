212  inline void MEP::AnimationObject::update(sf::Time& currentTime)
213  {
214  if (isRunning and (isInit == AnimationInit::ObjectAnimation)) {
215  if (currentTime - updateTime >= toWait) {
216  if (direction == Direction::Forward) {
217  if (currentFrame == --texture->end()) {
218  isRunning = false;
219  }
220  else {
221  currentFrame++;
222  index_currentFrame++;
223  }
224  }
225  else if (direction == Direction::Backwards) {
226  currentFrame--;
227  index_currentFrame++;
228  if (currentFrame == texture->begin()) {
229  isRunning = false;
230  }
231  }
232  updateTime = currentTime;
233  updateSprite(**currentFrame);
234  }
235  }
236  if (isInit == AnimationInit::Follow) {
237  if (toFollowOBJ->getInit() != AnimationInit::Follow) {
238  while (toFollowOBJ->index_currentFrame != index_currentFrame) {
239  if (toFollowOBJ->index_currentFrame > index_currentFrame) {
240  if (currentFrame == --texture->end()) {
241  break;
242  }
243  index_currentFrame++;
244  currentFrame++;
245  }
246  else {
247  if (currentFrame == texture->begin()) {
248  break;
249  }
250  index_currentFrame--;
251  currentFrame--;
252  }
253  }
254  if (toFollowOBJ->getPosition() != getPosition()) {
255  m_posFixed = toFollowOBJ->getPosition();
256  }
257  if (toFollowOBJ->getScale() != getScale()) {
258  m_scaleFixed = toFollowOBJ->getScale();
259  }
260  updateSprite(**currentFrame);
261  }
262  }
263  if (!followingList.empty()) {
264  updateSprite();
265  }
266  }
