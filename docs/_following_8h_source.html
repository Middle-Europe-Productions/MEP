75  void updateVariables(Following& object, sf::Sprite& toUpdate) {
76  if (m_followType == FollowType::X_Pos) {
77  object.m_posFixed.x = object.m_pos.x + m_animation.GetFixedVariable();
78  toUpdate.setPosition(object.m_posFixed);
79  }
80  else if (m_followType == FollowType::Y_Pos) {
81  object.m_posFixed.y = object.m_pos.y + m_animation.GetFixedVariable();
82  toUpdate.setPosition(object.m_posFixed);
83  }
84  else if (m_followType == FollowType::X_Scale) {
85  object.m_scaleFixed.x = object.m_scale.x + m_animation.GetFixedVariable();
86  toUpdate.setScale(object.m_scaleFixed);
87  }
88  else if (m_followType == FollowType::Y_Scale) {
89  object.m_scaleFixed.y = object.m_scale.y + m_animation.GetFixedVariable();
90  toUpdate.setScale(object.m_scaleFixed);
91  }
92  else if (m_followType == FollowType::Color) {
93  if (m_colorsFollow == MEP::ColorChannel::ALL)
94  toUpdate.setColor(m_animation.getFrameAsColor());
95  else {
96  if (m_colorsFollow != 0) {
97  sf::Uint8 RGBA[4];
98  if (m_colorsFollow & MEP::ColorChannel::R)
99  RGBA[0] = m_animation.getFrameAsColor().r;
100  else
101  RGBA[0] = toUpdate.getColor().r;
102  if (m_colorsFollow & MEP::ColorChannel::G)
103  RGBA[1] = m_animation.getFrameAsColor().g;
104  else
105  RGBA[1] = toUpdate.getColor().g;
106  if (m_colorsFollow & MEP::ColorChannel::B)
107  RGBA[2] = m_animation.getFrameAsColor().b;
108  else
109  RGBA[2] = toUpdate.getColor().b;
110  if (m_colorsFollow & MEP::ColorChannel::A)
111  RGBA[3] = m_animation.getFrameAsColor().a;
112  else
113  RGBA[3] = toUpdate.getColor().a;
114  toUpdate.setColor({ RGBA[0],RGBA[1],RGBA[2],RGBA[3] });

292  for (auto& x : followingList)
