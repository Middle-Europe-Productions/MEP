126  bool updateVariables(Following& object, sf::Sprite& toUpdate) {
127  if (!isLinked() or isMuted()) {
128  return false;
129  }
130  else if (m_followType == FollowType::X_Pos) {
131  object.m_posFixed.x = object.m_pos.x + m_animation->getFixedVariable();
132  toUpdate.setPosition(object.m_posFixed);
133  }
134  else if (m_followType == FollowType::Y_Pos) {
135  object.m_posFixed.y = object.m_pos.y + m_animation->getFixedVariable();
136  toUpdate.setPosition(object.m_posFixed);
137  }
138  else if (m_followType == FollowType::X_Scale) {
139  object.m_scaleFixed.x = object.m_scale.x + m_animation->getFixedVariable();
140  toUpdate.setScale(object.m_scaleFixed);
141  }
142  else if (m_followType == FollowType::Y_Scale) {
143  object.m_scaleFixed.y = object.m_scale.y + m_animation->getFixedVariable();
144  toUpdate.setScale(object.m_scaleFixed);
145  }
146  else if (m_followType == FollowType::Color) {
147  if (m_colorsFollow == MEP::ColorChannel::ALL)
148  toUpdate.setColor(m_animation->getFrameAsColor());
149  else {
150  if (m_colorsFollow != 0) {
151  sf::Uint8 RGBA[4];
152  if (m_colorsFollow & MEP::ColorChannel::R)
153  RGBA[0] = m_animation->getFrameAsColor().r;
154  else
155  RGBA[0] = toUpdate.getColor().r;
156  if (m_colorsFollow & MEP::ColorChannel::G)
157  RGBA[1] = m_animation->getFrameAsColor().g;
158  else
159  RGBA[1] = toUpdate.getColor().g;
160  if (m_colorsFollow & MEP::ColorChannel::B)
161  RGBA[2] = m_animation->getFrameAsColor().b;
162  else
163  RGBA[2] = toUpdate.getColor().b;
164  if (m_colorsFollow & MEP::ColorChannel::A)
165  RGBA[3] = m_animation->getFrameAsColor().a;
166  else
167  RGBA[3] = toUpdate.getColor().a;
168  toUpdate.setColor({ RGBA[0],RGBA[1],RGBA[2],RGBA[3] });
169  }
170  }
171  }
172  return true;
173  }
